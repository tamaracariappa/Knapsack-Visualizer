def brute_force_knapsack(weights, values, capacity):

n = len(weights)

A = [0] * n

best_value = 0

best_weight = 0

best_choice = [0] * n


for i in range(1, 2 ** n):

j = n - 1

temp_weight = 0

temp_value = 0


while j >= 0 and A[j] == 1:

A[j] = 0

j -= 1

if j >= 0:

A[j] = 1


for k in range(n):

if A[k] == 1:

temp_weight += weights[k]

temp_value += values[k]


if temp_value > best_value and temp_weight <= capacity:

best_value = temp_value

best_weight = temp_weight

best_choice = A[:]


included_items = [i + 1 for i in range(n) if best_choice[i] == 1]

print("Including item numbers:", included_items)

print("Total value:", best_value)

print("Total weight:", best_weight)


return best_choice, best_value, best_weight

import java.util.*;

public class Knapsack {

public static void bruteForceKnapsack(int[] weights, int[] values, int capacity) {

int n = weights.length;

int[] A = new int[n];

int bestValue = 0, bestWeight = 0;

int[] bestChoice = new int[n];

for (int i = 1; i < (1 << n); i++) {

int j = n - 1, tempWeight = 0, tempValue = 0;

while (j >= 0 && A[j] == 1) { A[j] = 0; j--; }

if (j >= 0) A[j] = 1;

for (int k = 0; k < n; k++)

if (A[k] == 1) { tempWeight += weights[k]; tempValue += values[k]; }

if (tempValue > bestValue && tempWeight <= capacity) {

bestValue = tempValue; bestWeight = tempWeight;

System.arraycopy(A, 0, bestChoice, 0, n);

}

}

List<Integer> items = new ArrayList<>();

for (int i = 0; i < n; i++) if (bestChoice[i] == 1) items.add(i + 1);

System.out.println("Including item numbers: " + items);

System.out.println("Total value: " + bestValue);

System.out.println("Total weight: " + bestWeight);

}

}

void bruteForceKnapsack(int* weights, int* values, int capacity, int n) {

int* A = (int*)malloc(n * sizeof(int));

int bestValue = 0, bestWeight = 0;

int* bestChoice = (int*)malloc(n * sizeof(int));

for (int i = 1; i < (1 << n); i++) {

int j = n - 1, tempWeight = 0, tempValue = 0;

while (j >= 0 && A[j] == 1) { A[j] = 0; j--; }

if (j >= 0) A[j] = 1;

for (int k = 0; k < n; k++) if (A[k] == 1) { tempWeight += weights[k]; tempValue += values[k]; }

if (tempValue > bestValue && tempWeight <= capacity) {

bestValue = tempValue; bestWeight = tempWeight;

for (int k = 0; k < n; k++) bestChoice[k] = A[k];

}

}

printf("Including item numbers: ");

for (int i = 0; i < n; i++) if (bestChoice[i] == 1) printf("%d ", i + 1);

printf("\nTotal value: %d\n", bestValue);
